"""
Self-Learning Module — Enhanced with per-specialty/band tracking,
temporal trend analysis, and question difficulty calibration.
Backward-compatible with existing weights.json and outcomes.jsonl.
"""

import json
import logging
import statistics
from datetime import datetime, timedelta
from pathlib import Path
from typing import Any, Dict, List, Optional, Tuple

logger = logging.getLogger(__name__)


class SelfLearning:
    """
    Enhanced self-learning engine that tracks outcomes per specialty and band,
    detects performance trends over time, and adjusts evaluation weights
    independently for each specialty/band combination.
    """

    DEFAULT_WEIGHTS: Dict[str, float] = {
        "knowledge_depth_weight": 1.0,
        "clinical_reasoning_weight": 1.0,
        "safety_awareness_weight": 1.0,
        "communication_weight": 1.0,
        "leadership_weight": 1.0,
    }

    def __init__(self, storage_dir: str = "models/self_learning") -> None:
        self.storage_path = Path(storage_dir)
        self.storage_path.mkdir(parents=True, exist_ok=True)

        # Global weights file (backward-compatible)
        self.weights_file = self.storage_path / "weights.json"
        if not self.weights_file.exists():
            self._save_weights(dict(self.DEFAULT_WEIGHTS))

        # Per-specialty/band weights directory
        self.granular_dir = self.storage_path / "granular_weights"
        self.granular_dir.mkdir(parents=True, exist_ok=True)

        # Question difficulty data
        self.difficulty_file = self.storage_path / "question_difficulty.json"
        if not self.difficulty_file.exists():
            self._save_json(self.difficulty_file, {})

        logger.info("SelfLearning initialized — storage: %s", self.storage_path)

    # ------------------------------------------------------------------
    # Low-level helpers
    # ------------------------------------------------------------------

    def _save_json(self, path: Path, data: Any) -> None:
        with open(path, "w", encoding="utf-8") as fh:
            json.dump(data, fh, indent=2, default=str)

    def _load_json(self, path: Path) -> Any:
        with open(path, "r", encoding="utf-8") as fh:
            return json.load(fh)

    def _save_weights(self, weights: Dict[str, float]) -> None:
        self._save_json(self.weights_file, weights)

    # ------------------------------------------------------------------
    # Public — weights
    # ------------------------------------------------------------------

    def get_weights(self, specialty: Optional[str] = None, band: Optional[str] = None) -> Dict[str, float]:
        """Return weights — granular if specialty/band provided, else global."""
        if specialty and band:
            key = f"{specialty}_{band}"
            gfile = self.granular_dir / f"{key}.json"
            if gfile.exists():
                return self._load_json(gfile)
        return self._load_json(self.weights_file)

    def _save_granular_weights(self, specialty: str, band: str, weights: Dict[str, float]) -> None:
        key = f"{specialty}_{band}"
        gfile = self.granular_dir / f"{key}.json"
        self._save_json(gfile, weights)

    # ------------------------------------------------------------------
    # Public — outcome recording (enhanced)
    # ------------------------------------------------------------------

    def record_outcome(self, outcome: Dict[str, Any]) -> None:
        """
        Record a training outcome.  Accepts the legacy format
        {"band", "specialty", "overall_score", "user_feedback_score"}
        as well as the enhanced format that adds:
          - detailed_scores  (dict of per-criterion scores)
          - question_id      (int)
          - time_spent_s     (float — seconds spent on the question)
          - timestamp        (ISO string — auto-filled if missing)
        """
        if "timestamp" not in outcome:
            outcome["timestamp"] = datetime.utcnow().isoformat()

        log_file = self.storage_path / "outcomes.jsonl"
        with open(log_file, "a", encoding="utf-8") as fh:
            fh.write(json.dumps(outcome) + "\n")

        # Update question difficulty if question_id is present
        qid = outcome.get("question_id")
        if qid is not None:
            self._update_question_difficulty(
                question_id=str(qid),
                score=float(outcome.get("overall_score", 0)),
                time_spent=outcome.get("time_spent_s"),
            )

        logger.debug("Outcome recorded — band=%s specialty=%s score=%s",
                      outcome.get("band"), outcome.get("specialty"),
                      outcome.get("overall_score"))

    # ------------------------------------------------------------------
    # Public — retrain (enhanced)
    # ------------------------------------------------------------------

    def retrain(self, specialty: Optional[str] = None, band: Optional[str] = None) -> Dict[str, float]:
        """
        Re-calculate weights.  If specialty and band are given, only outcomes
        matching that combination are used and a granular weights file is
        written.  Otherwise the global weights file is updated (backward-
        compatible behaviour).
        """
        outcomes = self._load_outcomes(specialty=specialty, band=band)
        if not outcomes:
            return self.get_weights(specialty, band)

        # Compute per-criterion averages
        criterion_keys = [
            "knowledge_depth", "clinical_reasoning",
            "safety_awareness", "communication", "leadership",
        ]
        criterion_totals: Dict[str, List[float]] = {k: [] for k in criterion_keys}

        for o in outcomes:
            ds = o.get("detailed_scores", {})
            for ck in criterion_keys:
                val = ds.get(ck)
                if val is not None:
                    criterion_totals[ck].append(float(val))
                else:
                    overall = float(o.get("overall_score", 0))
                    criterion_totals[ck].append(overall)

        weights = dict(self.DEFAULT_WEIGHTS)
        for ck in criterion_keys:
            wk = f"{ck}_weight"
            scores = criterion_totals[ck]
            if not scores:
                continue
            avg = statistics.mean(scores)
            # Increase weight when scores are low (area needs focus),
            # decrease when scores are high.
            factor = 1.0 + (75.0 - avg) / 300.0
            weights[wk] = max(0.5, min(2.0, weights.get(wk, 1.0) * factor))

        # Save
        if specialty and band:
            self._save_granular_weights(specialty, band, weights)
        else:
            self._save_weights(weights)
            # Also update each specialty/band combo found in the data
            combos = {(o.get("specialty"), o.get("band")) for o in outcomes
                      if o.get("specialty") and o.get("band")}
            for sp, bd in combos:
                self.retrain(specialty=sp, band=bd)

        logger.info("Retrained weights — specialty=%s band=%s weights=%s",
                     specialty, band, weights)
        return weights

    # ------------------------------------------------------------------
    # Public — trend analysis
    # ------------------------------------------------------------------

    def get_trends(self, specialty: Optional[str] = None, band: Optional[str] = None,
                   window_days: int = 7) -> Dict[str, Any]:
        """
        Return performance trends: compare recent window vs previous window.
        """
        outcomes = self._load_outcomes(specialty=specialty, band=band)
        if not outcomes:
            return {"status": "no_data"}

        now = datetime.utcnow()
        recent_cutoff = now - timedelta(days=window_days)
        prev_cutoff = recent_cutoff - timedelta(days=window_days)

        recent_scores: List[float] = []
        prev_scores: List[float] = []

        for o in outcomes:
            ts_str = o.get("timestamp")
            if not ts_str:
                continue
            try:
                ts = datetime.fromisoformat(ts_str)
            except (ValueError, TypeError):
                continue
            score = float(o.get("overall_score", 0))
            if ts >= recent_cutoff:
                recent_scores.append(score)
            elif ts >= prev_cutoff:
                prev_scores.append(score)

        recent_avg = statistics.mean(recent_scores) if recent_scores else None
        prev_avg = statistics.mean(prev_scores) if prev_scores else None

        trend = "stable"
        delta = 0.0
        if recent_avg is not None and prev_avg is not None:
            delta = recent_avg - prev_avg
            if delta > 3.0:
                trend = "improving"
            elif delta < -3.0:
                trend = "degrading"

        return {
            "specialty": specialty,
            "band": band,
            "window_days": window_days,
            "recent_avg": round(recent_avg, 2) if recent_avg else None,
            "previous_avg": round(prev_avg, 2) if prev_avg else None,
            "delta": round(delta, 2),
            "trend": trend,
            "recent_count": len(recent_scores),
            "previous_count": len(prev_scores),
        }

    # ------------------------------------------------------------------
    # Public — question difficulty
    # ------------------------------------------------------------------

    def get_question_difficulty(self) -> Dict[str, Any]:
        """Return the current question difficulty map."""
        return self._load_json(self.difficulty_file)

    def _update_question_difficulty(self, question_id: str, score: float,
                                     time_spent: Optional[float] = None) -> None:
        data = self._load_json(self.difficulty_file)
        entry = data.get(question_id, {"scores": [], "times": [], "attempts": 0})
        entry["scores"].append(score)
        if time_spent is not None:
            entry["times"].append(time_spent)
        entry["attempts"] = entry.get("attempts", 0) + 1
        entry["avg_score"] = round(statistics.mean(entry["scores"]), 2)
        if entry["times"]:
            entry["avg_time"] = round(statistics.mean(entry["times"]), 2)
        # Difficulty: 0 = very easy, 100 = very hard
        entry["difficulty"] = round(100.0 - entry["avg_score"], 2)
        data[question_id] = entry
        self._save_json(self.difficulty_file, data)

    # ------------------------------------------------------------------
    # Public — failure patterns
    # ------------------------------------------------------------------

    def detect_failure_patterns(self, threshold: float = 60.0) -> List[Dict[str, Any]]:
        """
        Scan outcomes for specialty/band combinations where the average
        score falls below *threshold*.  Returns a list of pattern dicts.
        """
        outcomes = self._load_outcomes()
        if not outcomes:
            return []

        buckets: Dict[str, List[float]] = {}
        for o in outcomes:
            key = f"{o.get('specialty', 'unknown')}|{o.get('band', 'unknown')}"
            buckets.setdefault(key, []).append(float(o.get("overall_score", 0)))

        patterns: List[Dict[str, Any]] = []
        for key, scores in buckets.items():
            avg = statistics.mean(scores)
            if avg < threshold:
                sp, bd = key.split("|", 1)
                patterns.append({
                    "specialty": sp,
                    "band": bd,
                    "avg_score": round(avg, 2),
                    "total_attempts": len(scores),
                    "failure_rate": round(
                        sum(1 for s in scores if s < threshold) / len(scores) * 100, 1
                    ),
                    "severity": "critical" if avg < 40 else "high" if avg < 55 else "medium",
                })
        patterns.sort(key=lambda p: p["avg_score"])
        return patterns

    # ------------------------------------------------------------------
    # Internal — load outcomes
    # ------------------------------------------------------------------

    def _load_outcomes(self, specialty: Optional[str] = None,
                       band: Optional[str] = None) -> List[Dict[str, Any]]:
        log_file = self.storage_path / "outcomes.jsonl"
        if not log_file.exists():
            return []
        results: List[Dict[str, Any]] = []
        with open(log_file, "r", encoding="utf-8") as fh:
            for line in fh:
                try:
                    obj = json.loads(line)
                except Exception:
                    continue
                if specialty and obj.get("specialty") != specialty:
                    continue
                if band and obj.get("band") != band:
                    continue
                results.append(obj)
        return results

    # ------------------------------------------------------------------
    # Public — summary
    # ------------------------------------------------------------------

    def get_summary(self) -> Dict[str, Any]:
        """High-level summary of the learning system state."""
        outcomes = self._load_outcomes()
        combos: Dict[str, int] = {}
        for o in outcomes:
            key = f"{o.get('specialty', '?')}|{o.get('band', '?')}"
            combos[key] = combos.get(key, 0) + 1

        return {
            "total_outcomes": len(outcomes),
            "specialty_band_combos": len(combos),
            "combos_detail": combos,
            "global_weights": self.get_weights(),
            "failure_patterns": self.detect_failure_patterns(),
        }


# Global singleton (backward-compatible)
self_learning = SelfLearning()
